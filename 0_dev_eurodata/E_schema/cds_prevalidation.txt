CREATE TABLE `eurodata`.`cds_prevalidation` (
  `date` DATE NOT NULL,
  `site` VARCHAR(126) NOT NULL,
  `region` VARCHAR(45) NULL,
  `carburant` VARCHAR(45) NOT NULL,
  `statut` VARCHAR(45) NULL,
  `date_synchro` DATETIME NULL,
  `origine` VARCHAR(45) NULL COMMENT 'origine des données (WEB ou App Mobile)',
  `dlv_BL` DECIMAL(5,2) NULL COMMENT 'Quantité livraison dans le BL',
  `dlv_mesure` DECIMAL(5,2) NULL,
  `ecart_dlv` DECIMAL(5,2) NULL COMMENT '= livraison_mesuree - livraison_BL',
  `ecart_dlv_mille` DECIMAL(5,2) NULL COMMENT '\'= ecart_livraison*1000/livraison_BL',
  `stock_init` DECIMAL(5,2) NULL,
  `stock_theor_fin` DECIMAL(5,2) NULL COMMENT 'stock théorique final = stock_initial + livraison - vente + test pompe\n',
  `stock_fin` DECIMAL(5,2) NULL COMMENT 'stock réel final',
  `diff` DECIMAL(5,2) NULL COMMENT 'différence (stock final - stock théorique finale)',
  `vente` DECIMAL(5,2) NULL,
  `test_pompe` DECIMAL(5,2) NULL,
  `ecart_vente` DECIMAL(5,2) NULL,
  `cumul_vente` DECIMAL(10,2) NULL,
  `cumul_ecart` DECIMAL(10,2) NULL,
  `pourcent_ecart` DECIMAL(5,2) NULL,
  `directe_dlv` DECIMAL(5,2) NULL,
  `date_extract` DATETIME COMMENT 'date et heure de l\'extraction des données\n\"d/m/Y H:M:S\"\n',
  `data_check` TINYINT NULL,
  PRIMARY KEY (`date`, `site`, `carburant`,`date_extract`))
COMMENT = 'Table pour cds_prevalidation Eurodata.';
